```cpp
class Solution {
public:
    vector<int> canSeePersonsCount(vector<int>& heights) {
        // 适合从右往左遍历
        int n = heights.size();
        std::vector<int> answer(n, 0);
        std::stack<int> stk;
        for (auto i = n - 1; i >= 0; --i) {
            int cnt = 0;
            while (!stk.empty() && heights[i] > heights[stk.top()]) {
                stk.pop();
                cnt += 1;
            }
            answer[i] = stk.empty() ? cnt : cnt + 1;
            stk.push(i);
        }
        return answer;
    }
};
```
