# [Description](https://leetcode.cn/problems/daily-temperatures/description/)

## Code

从左到右遍历，只在出栈的时候更新答案，栈内从顶到底单调递增

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> answer(n);
        std::stack<int> stk;
        // 从左到右，只在出栈的时候更新答案
        for (int i = 0; i < n; ++i) {
            while (!stk.empty() && temperatures[i] > temperatures[stk.top()]) {
                auto idx = stk.top();
                answer[idx] = i - idx;
                stk.pop();
            }
            stk.push(i);
        }
        return answer;
    }
};
```

也可以从右往左遍历，仍然维持栈内单调递增，但是更新答案是在入栈的时候

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> answer(n);
        std::stack<int> stk;
        // 从右向左，入栈的时候更新答案。相同温度取靠左的
        for (int i = n - 1; i >= 0; --i) {
            while (!stk.empty() && temperatures[i] >= temperatures[stk.top()]) {
                stk.pop();
            }
            answer[i] = stk.empty() ? 0 : stk.top() - i;
            stk.push(i);
        }
        return answer;
    }
};
```
