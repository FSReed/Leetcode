```cpp
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        std::deque<int> dq;
        std::vector<int> results;
        for (int i = 0; i < nums.size(); ++i) {
            while (!dq.empty() && dq.back() < nums[i]) {
                dq.pop_back();
            }
            dq.push_back(nums[i]);
            if (i >= k - 1) {
                // 更新答案
                results.push_back(dq.front());
                if (nums[i - k + 1] == dq.front()) {
                    dq.pop_front();
                }
            }
        }
        return results;
    }
};
```
