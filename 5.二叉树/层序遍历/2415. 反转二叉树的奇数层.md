# [Description](https://leetcode.cn/problems/reverse-odd-levels-of-binary-tree/description/)

## Code

先用层序写一遍：

```cpp
class Solution {
public:
    TreeNode* reverseOddLevels(TreeNode* root) {
        // BFS，如果奇数层，记录答案，在结束后对各个节点重新赋值
        queue<TreeNode*> Q;
        int count = 0;
        if (root) Q.push(root);
        while (!Q.empty()) {
            int size = Q.size();
            TreeNode* current;
            vector<TreeNode*> temp;
            for (int i = 0; i < size; i++) {
                current = Q.front();
                if (current->left) Q.push(current->left);
                if (current->right) Q.push(current->right);
                temp.push_back(current);
                Q.pop();
            }
            if (count % 2) {
                // 利用temp进行值反转
                int l = 0, r = size - 1;
                while (l <= r) {
                    std::swap(temp[l++]->val, temp[r--]->val);
                }
            }
            count++;
        }

        return root;
    }
};
```

想起来之前写DFS时候遇到过[对称二叉树](<../二叉树的递归/101. 对称二叉树.md>)，这里也可以用DFS做，代码更加简洁：

```cpp
class Solution {
public:
    TreeNode* reverseOddLevels(TreeNode* root) {
        // DFS
        dfs(root->left, root->right, true);
        return root;
    }

    void dfs(TreeNode* left, TreeNode* right, bool is_odd) {
        if (left == nullptr || right == nullptr) return;
        if (is_odd) swap(left->val, right->val);
        dfs(left->left, right->right, !is_odd);
        dfs(left->right, right->left, !is_odd);
    }
};
```
