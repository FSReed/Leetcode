# [Description](leetcode.cn/problems/diameter-of-binary-tree/description/)

## Code

中途更新答案，因为最长直径并不一定穿过root节点

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int diameterOfBinaryTree(TreeNode* root) {
        int ans = 0;
        std::function<int(TreeNode *)> dfs = [&](TreeNode *node) {
            if (node == nullptr) {
                return -1;
            }
            int l_max = dfs(node->left);
            int r_max = dfs(node->right);
            ans = std::max(ans, l_max + r_max + 2);
            return std::max(l_max, r_max) + 1;
        };
        dfs(root);
        return ans;
    }
};
```
