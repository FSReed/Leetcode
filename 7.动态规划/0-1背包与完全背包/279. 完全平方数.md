```cpp
class Solution {
public:
    int numSquares(int n) {
        std::vector<int> nums;
        for (int i = 0; i <= 100; i++) {
            nums.push_back(i * i);
        }

        // 状态转移方程: f[i + 1][t] = min(f[i][t], f[i + 1][t - nums[i + 1]] + 1)
        std::vector<int> f(n + 1, INT_MAX / 2);
        f[0] = 0;
        for (auto x : nums) {
            for (int t = x; t <= n ; t++) {
                f[t] = std::min(f[t], f[t - x] + 1);
            }
        }
        return f[n];
    }
};
```
