```cpp
class Solution {
public:
    int maxProfit(vector<int>& prices, int fee) {
        // 仍然用卖出作为一次交易的结束
        // f(i, not_hold) = max(f(i - 1, not_hold), f(i - 1, hold) + prices[i] - fee)
        // f(i, hold) = max(f(i - 1, hold), f(i - 1, not_hold) - prices[i])
        int f0, f1;
        f0 = 0;
        f1 = INT_MIN / 2;
        int n = prices.size();
        for (int i = 0; i < n; ++i) {
            int not_hold = std::max(f0, f1 + prices[i] - fee);
            int hold = std::max(f1, f0 - prices[i]);
            f0 = not_hold;
            f1 = hold;
        }
        return f0;
    }
};
```
