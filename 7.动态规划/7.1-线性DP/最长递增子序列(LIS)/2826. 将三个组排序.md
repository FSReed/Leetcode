# [Description](https://leetcode.cn/problems/sorting-three-groups/)

## Code

```cpp
class Solution {
public:
    int minimumOperations(vector<int>& nums) {
        // 类似LIS, 不过g数组是单调不减的，二分时候查找第一个>nums[i]的位置
        std::vector<int> g;
        for (auto &num : nums) {
            auto pos = std::upper_bound(g.begin(), g.end(), num);
            if (pos == g.end()) {
                g.push_back(num);
            } else {
                *pos = num;
            }
        }
        return nums.size() - g.size();
    }
};
```
